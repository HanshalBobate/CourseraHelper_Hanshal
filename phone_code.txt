cat > server.py << 'EOF'
import requests
import subprocess
import time

LAPTOP_IP = input("Enter laptop IP: ")
PORT = 5000

while True:
    try:
        res = requests.get(f"http://{LAPTOP_IP}:{PORT}/get").json()
        text = res.get("text", "")

        if text:
            subprocess.run(["termux-clipboard-set", text])
            print("Updated:", text[:50])

    except Exception as e:
        print("Error:", e)

    time.sleep(1)
EOF
python server.py
